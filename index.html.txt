  <input type="number" id="keyH" value="56">
  <button id="applySize">サイズ変更</button>
</div>

<script>
const canvas = document.getElementById("canvas");
const textArea = document.getElementById("textArea");
const keySelect = document.getElementById("keySelect");
const keyWInput = document.getElementById("keyW");
const keyHInput = document.getElementById("keyH");
const applyBtn = document.getElementById("applySize");

const TAP_THRESHOLD = 6;

const KEYS = [
  "w","e","r","t","y","u","i","o","p",
  "a","s","d","g","h","k",
  "z","b","n","m",
  "space","back"
];

const keyData = {};

KEYS.forEach((key, i) => {
  const el = document.createElement("div");
  el.className = "key";
  el.textContent = key === "space" ? "␣" : key === "back" ? "⌫" : key;

  const left = 10 + (i % 7) * 60;
  const top = 10 + Math.floor(i / 7) * 70;

  el.style.left = left + "px";
  el.style.top = top + "px";
  el.style.width = "56px";
  el.style.height = "56px";

  canvas.appendChild(el);

  keyData[key] = {
    el, left, top,
    w: 56, h: 56,
    origLeft: left,
    origTop: top
  };

  enableDrag(el, key);
  addSelectOption(key);
});

applyBtn.addEventListener("click", () => {
  const key = keySelect.value;
  if (!key) return;

  const d = keyData[key];
  d.w = Number(keyWInput.value);
  d.h = Number(keyHInput.value);

  d.el.style.width = d.w + "px";
  d.el.style.height = d.h + "px";

  d.origLeft = d.left;
  d.origTop = d.top;
});

function enableDrag(el, key) {
  let startX, startY, startLeft, startTop;
  let moved = false;

  el.addEventListener("pointerdown", e => {
    e.preventDefault();
    moved = false;
    startX = e.clientX;
    startY = e.clientY;
    const d = keyData[key];
    startLeft = d.left;
    startTop = d.top;
    el.setPointerCapture(e.pointerId);
  });

  el.addEventListener("pointermove", e => {
    const dx = e.clientX - startX;
    const dy = e.clientY - startY;

    if (Math.abs(dx) > TAP_THRESHOLD || Math.abs(dy) > TAP_THRESHOLD) {
      moved = true;
      const d = keyData[key];
      d.left = startLeft + dx;
      d.top = startTop + dy;
      el.style.left = d.left + "px";
      el.style.top = d.top + "px";
    }
  });

  el.addEventListener("pointerup", e => {
    el.releasePointerCapture(e.pointerId);
    const d = keyData[key];

    if (!moved) {
      if (key === "back") {
        textArea.value = textArea.value.slice(0, -1);
      } else if (key === "space") {
        textArea.value += " ";
      } else {
        textArea.value += key;
      }
      return;
    }

    let overlap = false;
    for (const k in keyData) {
      if (k === key) continue;
      const o = keyData[k];
      if (
        d.left < o.left + o.w &&
        d.left + d.w > o.left &&
        d.top < o.top + o.h &&
        d.top + d.h > o.top
      ) {
        overlap = true;
        break;
      }
    }

    if (overlap) {
      d.left = d.origLeft;
      d.top = d.origTop;
      el.style.left = d.left + "px";
      el.style.top = d.top + "px";
    } else {
      d.origLeft = d.left;
      d.origTop = d.top;
    }
  });
}

function addSelectOption(key) {
  const opt = document.createElement("option");
  opt.value = key;
  opt.textContent = key;
  keySelect.appendChild(opt);
}
